import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Panel de Publicación',
      theme: ThemeData(
        primarySwatch: Colors.cyan,
      ),
      home: const PublicationPanelDemo(),
    );
  }
}

class PublicationPanel extends StatelessWidget {
  const PublicationPanel({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(20),
      ),
      padding: const EdgeInsets.all(20),
      constraints: const BoxConstraints(
        maxWidth: 400,
      ),
      child: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          const Text(
            'Crear publicación',
            style: TextStyle(
              fontSize: 22,
              fontWeight: FontWeight.bold,
              color: Colors.black,
            ),
          ),
          const SizedBox(height: 18),
          GridView.count(
            shrinkWrap: true,
            physics: const NeverScrollableScrollPhysics(),
            crossAxisCount: 2,
            mainAxisSpacing: 14,
            crossAxisSpacing: 14,
            childAspectRatio: 0.95,
            children: [
              _buildCategoryButton(
                context,
                icon: Icons.shopping_bag_outlined,
                label: 'Productos',
                onTap: () {
                  print('Productos seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.description_outlined,
                label: 'Servicios',
                onTap: () {
                  print('Servicios seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.person_outline,
                label: 'Empleos',
                onTap: () {
                  print('Empleos seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.directions_car_outlined,
                label: 'Vehículos',
                onTap: () {
                  print('Vehículos seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.home_outlined,
                label: 'Propiedades',
                onTap: () {
                  print('Propiedades seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.pets_outlined,
                label: 'Mascotas',
                onTap: () {
                  print('Mascotas seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.store_outlined,
                label: 'Alquiler',
                onTap: () {
                  print('Alquiler seleccionado');
                },
              ),
              _buildCategoryButton(
                context,
                icon: Icons.handshake_outlined,
                label: 'Trueques',
                onTap: () {
                  print('Trueques seleccionado');
                },
              ),
            ],
          ),
        ],
      ),
    );
  }

  Widget _buildCategoryButton(
    BuildContext context, {
    required IconData icon,
    required String label,
    required VoidCallback onTap,
  }) {
    return InkWell(
      onTap: onTap,
      borderRadius: BorderRadius.circular(28),
      child: Container(
        decoration: BoxDecoration(
          border: Border.all(
            color: const Color(0xFF4DD0E1),
            width: 2,
          ),
          borderRadius: BorderRadius.circular(28),
        ),
        padding: const EdgeInsets.symmetric(vertical: 16, horizontal: 12),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Container(
              width: 60,
              height: 60,
              decoration: BoxDecoration(
                color: const Color(0xFFDDDDDD),
                borderRadius: BorderRadius.circular(16),
              ),
              child: Icon(
                icon,
                size: 36,
                color: Colors.black,
              ),
            ),
            const SizedBox(height: 10),
            Text(
              label,
              style: const TextStyle(
                fontSize: 15,
                fontWeight: FontWeight.w500,
                color: Colors.black,
              ),
            ),
          ],
        ),
      ),
    );
  }
}

// Ejemplo de uso en una pantalla o modal
class PublicationPanelDemo extends StatelessWidget {
  const PublicationPanelDemo({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.grey[200],
      body: Center(
        child: Padding(
          padding: const EdgeInsets.all(16.0),
          child: PublicationPanel(),
        ),
      ),
    );
  }
}

// Para mostrar como modal bottom sheet
void showPublicationPanel(BuildContext context) {
  showModalBottomSheet(
    context: context,
    backgroundColor: Colors.transparent,
    isScrollControlled: true,
    builder: (context) => Padding(
      padding: const EdgeInsets.all(16.0),
      child: PublicationPanel(),
    ),
  );
}
